114 void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
115  auto x = H[i];
116  while (i > 1 && *x < *H[parent(i)]) {
117  set(i, H[parent(i)]);
118  i = parent(i);
119  }
120  set(i, x);
121 }

123 template <class T>
124 void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
125  auto x = H[i];
126  while (true) {
127  unsigned k = leftChild(i);
128  if (k >= H.size())
129  break;
130  if (k+1 < H.size() && *H[k+1] < *H[k])
131  ++k; // right child of i
132  if ( ! (*H[k] < *x) )
133  break;
134  set(i, H[k]);
135  i = k;
136  }
137  set(i, x);
138 }
